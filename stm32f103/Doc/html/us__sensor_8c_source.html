90 void ultrasonic_exti_callback (uint16_t GPIO_Pin) {
91  // Store the time to which the echo pin goes high
92  static uint64_t time_tmp[ULTRASONIC_NB] = {0};
93  // Store current state of the ultrasonic echo pin
94  static Ultrasonic_State state[ULTRASONIC_NB] = {DOWN};
95  // Store the echo pin which raised the current interruption
96  Ultrasonic_Position position = get_ultrasonic_position(GPIO_Pin);
97 
98  // If the echo pin was previously DOWN, update the current state to UP and save the time when it happened
99  if(state[position] == DOWN) {
100  state[position] = UP;
101  time_tmp[position] = micros();
102  }
103  // If the echo pin was previously UP, update the current state to DOWN and calculate the time during which the pulse was high
104  else{
105  state[position] = DOWN;
106  pulse_length[position] = micros() - time_tmp[position];
107  ultrasonic_callback(position);
108  }
109 }
